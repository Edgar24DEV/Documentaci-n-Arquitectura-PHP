Aggregate Root con invariantes de dominio

Order/Domain/Order.php — Aggregate Root 
final class Order
{
    private array       $items  = [];
    private array       $domainEvents = [];
    private OrderStatus $status;

    private function __construct(
        public readonly OrderId    $id,
        public readonly CustomerId $customerId,
        public readonly Currency   $currency,
    ) {
        $this->status = OrderStatus::pending();
    }

    public static function create(CustomerId $customerId, Currency $currency): self
    {
        $order = new self(OrderId::generate(), $customerId, $currency);
        $order->record(OrderCreated::create($order->id, $customerId));
        return $order;
    }

    public function addItem(Product $product, Quantity $quantity): void
    {
        $this->guardIsEditable();  // Guard con naming estándar
        $this->items[] = OrderItem::create($product, $quantity);
    }

    public function confirm(): void
    {
        $this->guardIsEditable();
        $this->guardHasItems();
        $this->status = OrderStatus::confirmed();
        $this->record(OrderConfirmed::create($this->id));
    }

    public function cancel(): void
    {
        if ($this->status->isShipped())
            throw new BusinessRuleViolationException(
                'Order cannot be cancelled after shipment',
                ['order_id' => $this->id->value()]
            );
        $this->status = OrderStatus::cancelled();
        $this->record(OrderCancelled::create($this->id));
    }

    // ═══ GUARDS — métodos privados para invariantes ═══
    private function guardIsEditable(): void
    {
        if (!$this->status->isEditable())
            throw new BusinessRuleViolationException(
                'Order is not in an editable state',
                ['status' => $this->status->value()]
            );
    }

    private function guardHasItems(): void
    {
        if (empty($this->items))
            throw new BusinessRuleViolationException('Order must have at least one item');
    }

    // ═══ DOMAIN EVENTS ═══
    /** @return DomainEvent[] */
    public function pullDomainEvents(): array
    {
        $events = $this->domainEvents;
        $this->domainEvents = [];
        return $events;
    }

    private function record(DomainEvent $event): void
    {
        $this->domainEvents[] = $event;
    }
}

Domain Events — Estructura base

💡
Todo Domain Event debe tener: eventId (UUID único), occurredOn (timestamp), y un método eventName(). Son inmutables y representan algo que ya ocurrió.
Shared/Domain/Event/DomainEvent.php 
abstract class DomainEvent
{
    private readonly string             $eventId;
    private readonly \DateTimeImmutable $occurredOn;

    protected function __construct()
    {
        $this->eventId    = Uuid::generate()->value();
        $this->occurredOn = new \DateTimeImmutable();
    }

    public function eventId(): string { return $this->eventId; }
    public function occurredOn(): \DateTimeImmutable { return $this->occurredOn; }
    abstract public function eventName(): string;
}

// Evento concreto — inmutable, datos mínimos
final class OrderConfirmed extends DomainEvent
{
    private function __construct(
        public readonly OrderId $orderId,
    ) {
        parent::__construct();
    }

    public static function create(OrderId $orderId): self
    {
        return new self($orderId);
    }

    public function eventName(): string
    {
        return 'order.confirmed';
    }
}
