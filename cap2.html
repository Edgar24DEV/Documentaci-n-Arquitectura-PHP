S — Single Responsibility en Laravel

OrderController.php — 4 responsabilidades 
class OrderController extends Controller
{
    public function store(Request $request)
    {
        // ① Validación (debería ser un Form Request)
        if (empty($request->items)) {
            return response()->json(['error' => 'Sin items'], 422);
        }
        // ② Lógica de negocio (debería ser un Use Case)
        $total = array_sum(array_column($request->items, 'price'));
        if ($request->coupon === 'VIP20') $total *= 0.8;
        // ③ Persistencia (debería ser un Repository)
        $order = Order::create(['total' => $total, 'user_id' => auth()->id()]);
        // ④ Notificación (debería ser un NotificationService)
        Mail::to(auth()->user()->email)->send(OrderConfirmed::create($order));
        return response()->json($order, 201);
    }
}
OrderController.php — solo HTTP 
class OrderController extends Controller
{
    public function store(
        StoreOrderRequest $request,    // ① valida
        PlaceOrderAction  $action,
    ) {
        $order = $action->execute(
            PlaceOrderDTO::fromRequest($request)  // ② + ③ + ④ encapsulados
        );
        return OrderResource::create($order);  // ⑤ formatea
    }
}
// Cada clase: UNA razón para cambiar
// StoreOrderRequest → reglas de validación
// PlaceOrderAction  → lógica de negocio
// OrderRepository   → persistencia
// OrderNotifier     → proveedor email

D — Dependency Inversion

DIP completo — interfaz, implementación y wiring 
// ① Abstracción — vive en el Dominio (cero deps de Laravel)
interface OrderRepository
{
    public function findById(OrderId $id): ?Order;
    public function save(Order $order): void;
}

// ② Implementación concreta — vive en Infraestructura
final class EloquentOrderRepository implements OrderRepository
{
    public function findById(OrderId $id): ?Order
    {
        $model = OrderModel::find($id->value());
        return $model ? OrderMapper::toDomain($model) : null;
    }
    public function save(Order $order): void
    {
        OrderModel::updateOrCreate(
            ['id' => $order->id->value()],
            OrderMapper::toPersistence($order)
        );
    }
}

// ③ Wiring — AppServiceProvider.php
$this->app->bind(OrderRepository::class, EloquentOrderRepository::class);
// En tests: swap sin tocar nada más
